type shapeType =  None | Rect | RoundedRect | Ellipse | Edge;

interface Root { 
  input radius : int;
  input centerRadius : int;
  input centerAlpha : int;
  var render : int;
  
  input w : int;
  input h : int;
  input bgcolor : color;
}
interface Node { 
  input open : float;
  var show : float;

  var r : int;
  var alpha : int;
  var sectorAng : int;
  var maxR : int;
  var level : int;
  
  var x : int;
  var y : int;
  var w : int;
  var h : int;
  var parentX : float;
  var parentY : float;
   
  var render : int;
  input bgcolor : color;
  
}


class Top : Root { 
  children { 
    child : Node;
  }
  constraints {
    child.alpha := 45;
    child.maxR := radius;        
    child.sectorAng := 360;

    child.level := 1;
    
    child.show := 1.0;
    
    child.parentX := (centerRadius/2 + child.w/2) * cos(PI() * centerAlpha / 180) - child.w/2;
    child.parentY := (centerRadius/2 + child.w/2) * cos(PI() * centerAlpha / 180) - child.w/2;
    
    render := paintStart(w, h) + paint(Rect, 0, 0, w, h, bgcolor);
  }  
}

class Radial : Node { 
  children {
    child : [ Node ];
  } 
  attributes {
    var numOpenChildren : int;
    
    var xOpen : int;
    var yOpen : int;
    var xClose : int;
    var yClose : int;    
  }
  constraints {
  
    r := maxR / pow(1.8, level);
  
    loop child {
      child.show := show * child.open;
    
      child.maxR := maxR;    
      
      child.sectorAng := sectorAng / $$.numOpenChildren;
      child.alpha := 
          fold alpha - (sectorAng / 2) - (sectorAng/($$.numOpenChildren*2)) 
          .. 
          child$-.alpha + child$i.show * (sectorAng/$$.numOpenChildren);

      child.level := level + 1;
      numOpenChildren := fold 0 .. $-.numOpenChildren + child$i.show;
      
      child.parentX := x;
      child.parentY := y;
        
      render := 
        fold 
          paint(Ellipse, x, y, w, h, bgcolor)
          ..
          paint(Edge, x, y, child$i.x, child$i.y, bgcolor);
        
    }
    w := pow(show,level) * 3.0 * r / 8.0;
    h := w;
    
    xOpen := parentX + r * cos(PI() * alpha / 180);
    yOpen := parentY + r * sin(PI() * alpha / 180);      
    xClose := parentX;
    yClose := parentY;
    x := show * xOpen + (1 - show) * xClose;
    y := show * yOpen + (1 - show) * yClose;
  } 
}
